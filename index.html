// ==================== FLUX PROXY (AMAN) ==================== async function generateWithFlux(prompt) { const btn = document.getElementById('generateBtn'); const originalHTML = btn.innerHTML; btn.innerHTML = ` Generating with Flux...`; btn.disabled = true; try { const response = await fetch('/api/generate-flux', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt }) }); const result = await response.json(); if (!response.ok) throw new Error(result.error || 'Failed to generate'); showFluxResult(result.imageUrl, prompt); } catch (err) { alert("❌ Gagal: " + err.message); console.error(err); } finally { btn.innerHTML = originalHTML; btn.disabled = false; } } // Tampilkan hasil (sama seperti sebelumnya) function showFluxResult(imageUrl, prompt) { const resultHTML = `

✨ Desain Flux Berhasil!

"${prompt}"

`; document.getElementById('modalContent').innerHTML = resultHTML; } // Panggil fungsi function generateAI() { const prompt = document.getElementById('promptInput').value.trim(); if (!prompt) { alert("Silakan tulis deskripsi baju impianmu 🌸"); return; } generateWithFlux(prompt); }
